\begin{problem}{A + B Problem}{标准输入}{标准输出}{1 s}{256 MB}
    给定两个整数 $a$ 和 $b$，求 $a + b$ 的值。

    \InputFile
    输入两个整数 $a$ 和 $b$。

    \OutputFile
    输出 $a + b$ 的值。

    \Example

    \begin{example}
    \exmp{
        1 2
        }{
        3
        } %
    \end{example}

    \Constraints
    $0 \leq a, b \leq 10^9$

    \Explanations
    在示例 1 中：
    输入 $1$ 和 $2$，输出 $3$。
\end{problem}
